
DYNAMIC IMPORTING TACTIC LIBRARIES

import importlib

def load_tactic_library(library_name: str):
    # Map Lean naming convention to Python file structure
    module_path = f"sudoku_prover_app.tactics.{library_name.lower()}"
    try:
        # This executes the @registry.register decorators in that file
        importlib.import_module(module_path)
    except ImportError:
        print(f"Warning: No Python tactics found for {library_name}")
