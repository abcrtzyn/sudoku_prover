from typing import Callable, Dict, Any, List, Type
from dataclasses import dataclass

@dataclass
class CommandEntry:
    func: Callable
    arg_types: List[Type]  # e.g., [int, str]

class CommandRegistry:
    def __init__(self):
        self.commands: Dict[str, CommandEntry] = {}

    def register(self, name: str, arg_types: List[Type]):
        """Decorator to register a tactic with its expected argument types."""
        def decorator(func: Callable):
            self.commands[name] = CommandEntry(func, arg_types)
            return func
        return decorator

# Create a single global instance
registry = CommandRegistry()

from .registry import registry
# Note: You do NOT import basic_tactics here to avoid the circle!

class ProofEngine:
    def execute_command(self, cmd_name: str, raw_args: List[str]):
        if cmd_name not in registry.commands:
            raise ValueError(f"Unknown command: {cmd_name}")

        entry = registry.commands[cmd_name]
        
        # 1. Check argument count
        if len(raw_args) != len(entry.arg_types):
            raise ValueError(f"{cmd_name} expects {len(entry.arg_types)} args")

        # 2. Automatically convert types (e.g., "5" -> 5)
        try:
            parsed_args = [
                t(arg) for t, arg in zip(entry.arg_types, raw_args)
            ]
        except ValueError:
            raise ValueError(f"Invalid argument types for {cmd_name}")

        # 3. Run it!
        # Since the tactic was decorated with @command_flow, 
        # it returns the generator we need.
        return entry.func(self, *parsed_args)
